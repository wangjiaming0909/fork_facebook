378 void
379 HTTPSession::readTimeoutExpired() noexcept {
380  VLOG(3) << "session-level timeout on " << *this;
381 
382  if (liveTransactions_ != 0) {
383  // There's at least one open transaction with a read timeout scheduled.
384  // We got here because the session timeout == the transaction timeout.
385  // Ignore, since the transaction is going to timeout very soon.
386  VLOG(4) << *this <<
387  "ignoring session timeout, transaction timeout imminent";
388  resetTimeout();
389  return;
390  }
391 
392  if (!transactions_.empty()) {
393  // There are one or more transactions, but none of them are live.
394  // That's valid if they've all received their full ingress messages
395  // and are waiting for their Handlers to process those messages.
396  VLOG(4) << *this <<
397  "ignoring session timeout, no transactions awaiting reads";
398  resetTimeout();
399  return;
400  }
401 
402  VLOG(4) << *this << " Timeout with nothing pending";
403 
404  setCloseReason(ConnectionCloseReason::TIMEOUT);
405  auto controller = getController();
406  if (controller) {
407  timeout_.scheduleTimeout(&drainTimeout_,
408  controller->getGracefulShutdownTimeout());
409  }
410  notifyPendingShutdown();
411 }

574 void
575 HTTPSession::processReadData() {
576  FOLLY_SCOPED_TRACE_SECTION("HTTPSession - processReadData");
577  // skip any empty IOBufs before feeding CODEC.
578  while (readBuf_.front() != nullptr && readBuf_.front()->length() == 0) {
579  readBuf_.pop_front();
580  }
581 
582  // Pass the ingress data through the codec to parse it. The codec
583  // will invoke various methods of the HTTPSession as callbacks.
584  const IOBuf* currentReadBuf;
585  // It's possible for the last buffer in a chain to be empty here.
586  // AsyncTransport saw fd activity so asked for a read buffer, but it was
587  // SSL traffic and not enough to decrypt a whole record. Later we invoke
588  // this function from the loop callback.
589  while (!ingressError_ &&
590  readsUnpaused() &&
591  ((currentReadBuf = readBuf_.front()) != nullptr &&
592  currentReadBuf->length() != 0)) {
593  // We're about to parse, make sure the parser is not paused
594  codec_->setParserPaused(false);
595  size_t bytesParsed = codec_->onIngress(*currentReadBuf);
596  if (bytesParsed == 0) {
597  // If the codec didn't make any progress with current input, we
598  // better get more.
599  break;
600  }
601  readBuf_.trimStart(bytesParsed);
602  }
603 }

757 void
758 HTTPSession::onMessageBegin(HTTPCodec::StreamID streamID, HTTPMessage* msg) {
759  VLOG(4) << "processing new msg streamID=" << streamID << " " << *this;
760  if (infoCallback_) {
761  infoCallback_->onRequestBegin(*this);
762  }
763 
764  HTTPTransaction* txn = findTransaction(streamID);
765  if (txn) {
766  if (isDownstream() && txn->isPushed()) {
767  // Push streams are unidirectional (half-closed). If the downstream
768  // attempts to send ingress, abort with STREAM_CLOSED error.
769  HTTPException ex(HTTPException::Direction::INGRESS_AND_EGRESS,
770  "Downstream attempts to send ingress, abort.");
771  ex.setCodecStatusCode(ErrorCode::STREAM_CLOSED);
772  txn->onError(ex);
773  }
774  return; // If this transaction is already registered, no need to add it now
775  }
776 
777  http2::PriorityUpdate messagePriority = getMessagePriority(msg);
778  txn = createTransaction(streamID, HTTPCodec::NoStream,
779  HTTPCodec::NoExAttributes, messagePriority);
780  if (!txn) {
781  return; // This could happen if the socket is bad.
782  }
783 
784  if (!codec_->supportsParallelRequests() && getPipelineStreamCount() > 1) {
785  // The previous transaction hasn't completed yet. Pause reads until
786  // it completes; this requires pausing both transactions.
787 
788  // HTTP/1.1 pipeline is detected, and which is incompactible with
789  // ByteEventTracker. Drain all the ByteEvents
790  CHECK(byteEventTracker_);
791  byteEventTracker_->drainByteEvents();
792 
793  // drainByteEvents() may detach txn(s). Don't pause read if one txn left
794  if (getPipelineStreamCount() < 2) {
795  DCHECK(readsUnpaused());
796  return;
797  }
798 
799  // There must be at least two transactions (we just checked). The previous
800  // txns haven't completed yet. Pause reads until they complete
801  DCHECK_GE(transactions_.size(), 2);
802  for (auto it = ++transactions_.rbegin(); it != transactions_.rend(); ++it) {
803  DCHECK(it->second.isIngressEOMSeen());
804  it->second.pauseIngress();
805  }
806  transactions_.rbegin()->second.pauseIngress();
807  DCHECK_EQ(liveTransactions_, 0);
808  DCHECK(readsPaused());
809  }

900  unique_ptr<HTTPMessage> msg) {

902  // headers.
903  VLOG(4) << "processing ingress headers complete for " << *this <<
904  ", streamID=" << streamID;
905 
906  if (!codec_->isReusable()) {
907  setCloseReason(ConnectionCloseReason::REQ_NOTREUSABLE);
908  }

911  infoCallback_->onIngressMessage(*this, *msg.get());
912  }

920  transportInfo_.sslCipher ? transportInfo_.sslCipher->c_str() : nullptr;

993 void HTTPSession::onChunkHeader(HTTPCodec::StreamID streamID,
994  size_t length) {
995  // The codec's parser detected a chunk header (meaning that this
996  // connection probably is HTTP/1.1).
997  //
998  // After calling onChunkHeader(), the codec will call onBody() zero
999  // or more times and then call onChunkComplete().
1000  //
1001  // The reason for this callback on the chunk header is to support
1002  // an optimization. In general, the job of the codec is to present
1003  // the HTTPSession with an abstract view of a message,
1004  // with all the details of wire formatting hidden. However, there's
1005  // one important case where we want to know about chunking: reverse
1006  // proxying where both the client and server streams are HTTP/1.1.
1007  // In that scenario, we preserve the server's chunk boundaries when
1008  // sending the response to the client, in order to avoid possibly
1009  // making the egress packetization worse by rechunking.
1010  HTTPTransaction* txn = findTransaction(streamID);
1011  if (!txn) {
1012  invalidStream(streamID);
1013  return;
1014  }
1015  txn->onIngressChunkHeader(length);

1040 void
1041 HTTPSession::onMessageComplete(HTTPCodec::StreamID streamID,
1042  bool upgrade) {
1043  DestructorGuard dg(this);
1044  // The codec's parser detected the end of the ingress message for
1045  // this transaction.
1046  VLOG(4) << "processing ingress message complete for " << *this <<
1047  ", streamID=" << streamID;
1048  HTTPTransaction* txn = findTransaction(streamID);
1049  if (!txn) {
1050  invalidStream(streamID);
1051  return;
1052  }
1053 
1054  if (upgrade) {
1055  /* Send the upgrade callback to the transaction and the handler.
1056  * Currently we support upgrades for only HTTP sessions and not SPDY
1057  * sessions.
1058  */
1059  ingressUpgraded_ = true;
1060  txn->onIngressUpgrade(UpgradeProtocol::TCP);
1061  return;
1062  }
1063 
1064  // txnIngressFinished = !1xx response
1065  const bool txnIngressFinished =
1066  txn->isDownstream() || !txn->extraResponseExpected();
1067  if (txnIngressFinished) {
1068  decrementTransactionCount(txn, true, false);
1069  }
1070  txn->onIngressEOM();
1071 
1072  // The codec knows, based on the semantics of whatever protocol it
1073  // supports, whether it's valid for any more ingress messages to arrive
1074  // after this one. For example, an HTTP/1.1 request containing
1075  // "Connection: close" indicates the end of the ingress, whereas a
1076  // SPDY session generally can handle more messages at any time.
1077  //
1078  // If the connection is not reusable, we close the read side of it
1079  // but not the write side. There are two reasons why more writes
1080  // may occur after this point:
1081  // * If there are previous writes buffered up in the pendingWrites_
1082  // queue, we need to attempt to complete them.
1083  // * The Handler associated with the transaction may want to
1084  // produce more egress data when the ingress message is fully
1085  // complete. (As a common example, an application that handles
1086  // form POSTs may not be able to even start generating a response
1087  // until it has received the full request body.)
1088  //
1089  // There may be additional checks that need to be performed that are
1090  // specific to requests or responses, so we call the subclass too.
1091  if (!codec_->isReusable() &&
1092  txnIngressFinished &&
1093  !codec_->supportsParallelRequests()) {
1094  VLOG(4) << *this << " cannot reuse ingress";
1095  shutdownTransport(true, false);
1096  }
1097 }

1200  ErrorCode code,
1201  std::unique_ptr<folly::IOBuf> debugData) {

1204

1206

1208  drain();

1213  invokeOnAllTransactions(&HTTPTransaction::onGoaway, code);

1216  // successfully at the remote end. Upstream transactions are created
1217  // with odd transaction IDs and downstream transactions with even IDs.

1219  auto firstStream = HTTPCodec::NoStream;

1228  // we can determine whether we have a codec error code.
1229  firstStream = streamID;
1230  continue;
1231  }
1232 
1233  ids.push_back(streamID);
1234  }
1235  }
1236 
1237 
1238  if (firstStream != HTTPCodec::NoStream && code != ErrorCode::NO_ERROR) {

1240  // by delivering a specific error to it and let the rest of the streams
1241  // get a normal unacknowledged stream error.
1242  ProxygenError err = kErrorStreamUnacknowledged;
1243  string debugInfo = (debugData) ?
1244  folly::to<string>(" with debug info: ", (char*)debugData->data()) : "";

1247  " on transaction id: ", *firstStream,

1249  debugInfo));
1250  ex.setProxygenError(err);
1251  errorOnTransactionId(*firstStream, std::move(ex));
1252  } else if (firstStream != HTTPCodec::NoStream) {
1253  ids.push_back(*firstStream);
1254  }
1255 
1256  errorOnTransactionIds(ids, kErrorStreamUnacknowledged);
1257 }

1585  const HTTPMessage& headers,
1586  HTTPHeaderSize* size,
1587  bool includeEOM) noexcept {

1620  txn->getID(),
1621  headers,
1622  *assocStream,
1623  includeEOM,

1627  txn->getID(),
1628  headers,
1629  includeEOM,
1630  size);
1631  }

1635  if (isDownstream() && headers.isResponse() && txn->isPushed()) {
1636  incrementOutgoingStreams();
1637  }
1638

1642  (isDownstream() && headers.isResponse());

1721 void
1722 HTTPSession::onEgressMessageFinished(HTTPTransaction* txn, bool withRST) {
1723  // If the semantics of the protocol don't permit more messages
1724  // to be read or sent on this connection, close the socket in one or
1725  // more directions.
1726  CHECK(!transactions_.empty());
1727 
1728  if (infoCallback_) {
1729  infoCallback_->onRequestEnd(*this, txn->getMaxDeferredSize());
1730  }
1731  auto oldStreamCount = getPipelineStreamCount();
1732  decrementTransactionCount(txn, false, true);
1733  if (withRST || ((!codec_->isReusable() || readsShutdown()) &&
1734  transactions_.size() == 1)) {
1735  // We should shutdown reads if we are closing with RST or we aren't
1736  // interested in any further messages (ie if we are a downstream session).
1737  // Upgraded sessions have independent ingress and egress, and the reads
1738  // need not be shutdown on egress finish.
1739  if (withRST) {
1740  // Let any queued writes complete, but send a RST when done.
1741  VLOG(4) << *this << " resetting egress after this message";
1742  resetAfterDrainingWrites_ = true;
1743  setCloseReason(ConnectionCloseReason::TRANSACTION_ABORT);
1744  shutdownTransport(true, true);
1745  } else {
1746  // the reason is already set (either not reusable or readshutdown).
1747 
1748  // Defer normal shutdowns until the end of the loop. This
1749  // handles an edge case with direct responses with Connection:
1750  // close served before ingress EOM. The remainder of the ingress
1751  // message may be in the parse loop, so give it a chance to
1752  // finish out and avoid a kErrorEOF
1753 
1754  // we can get here during shutdown, in that case do not schedule a
1755  // shutdown callback again
1756  if (!shutdownTransportCb_) {
1757  // Just for safety, the following bumps the refcount on this session
1758  // to keep it live until the loopCb runs
1759  shutdownTransportCb_.reset(new ShutdownTransportCallback(this));
1760  sock_->getEventBase()->runInLoop(shutdownTransportCb_.get(), true);
1761  }
1762  }
1763  } else {
1764  maybeResumePausedPipelinedTransaction(oldStreamCount,
1765  txn->getSequenceNumber());
1766  }
1767 }

1906  HTTPCodec::StreamID streamID = txn->getID();
1907  auto txnSeqn = txn->getSequenceNumber();
1908  auto it = transactions_.find(txn->getID());
1909  DCHECK(it != transactions_.end());
1910

1912  // Someone detached a transaction that was paused. Make the resumeIngress
1913  // call to keep liveTransactions_ in order
1914  VLOG(4) << *this << " detached paused transaction=" << streamID;
1915  resumeIngress(txn);
1916  }
1917 
1918  VLOG(4) << *this << " removing streamID=" << streamID <<
1919  ", liveTransactions was " << liveTransactions_;
1920  CHECK_GT(liveTransactions_, 0);
1921  liveTransactions_--;
1922 
1923  if (txn->isPushed()) {
1924  auto assocTxn = findTransaction(*txn->getAssocTxnId());
1925  if (assocTxn) {
1926  assocTxn->removePushedTransaction(streamID);
1927  }
1928  }
1929  if (txn->getControlStream()) {
1930  auto controlTxn = findTransaction(*txn->getControlStream());
1931  if (controlTxn) {
1932  controlTxn->removeExTransaction(streamID);
1933  }
1934  }
1935

1937  decrementTransactionCount(txn, true, true);
1938  transactions_.erase(it);
1939 
1940  if (transactions_.empty()) {
1941  HTTPSessionBase::setLatestActive();
1942  if (infoCallback_) {
1943  infoCallback_->onDeactivateConnection(*this);
1944  }
1945  if (getConnectionManager()) {
1946  getConnectionManager()->onDeactivated(*this);
1947  }
1948  } else {
1949  if (infoCallback_) {
1950  infoCallback_->onTransactionDetached(*this);
1951  }
1952  }
1953

1955  if (maybeResumePausedPipelinedTransaction(oldStreamCount, txnSeqn)) {
1956  return;

2055 unique_ptr<IOBuf> HTTPSession::getNextToSend(bool* cork, bool* eom) {
2056  // limit ourselves to one outstanding write at a time (onWriteSuccess calls
2057  // scheduleWrite)
2058  if (numActiveWrites_ > 0 || writesShutdown()) {
2059  VLOG(4) << "skipping write during this loop, numActiveWrites_=" <<
2060  numActiveWrites_ << " writesShutdown()=" << writesShutdown();
2061  return nullptr;
2062  }
2063 
2064  // We always tack on at least one body packet to the current write buf
2065  // This ensures that a short HTTPS response will go out in a single SSL record
2066  while (!txnEgressQueue_.empty()) {
2067  uint32_t toSend = kWriteReadyMax;
2068  if (connFlowControl_) {
2069  if (connFlowControl_->getAvailableSend() == 0) {
2070  VLOG(4) << "Session-level send window is full, skipping remaining "
2071  << "body writes this loop";
2072  break;
2073  }
2074  toSend = std::min(toSend, connFlowControl_->getAvailableSend());
2075  }
2076  txnEgressQueue_.nextEgress(nextEgressResults_,
2077  isSpdyCodecProtocol(codec_->getProtocol()));
2078  CHECK(!nextEgressResults_.empty()); // Queue was non empty, so this must be
2079  // The maximum we will send for any transaction in this loop
2080  uint32_t txnMaxToSend = toSend * nextEgressResults_.front().second;
2081  if (txnMaxToSend == 0) {
2082  // toSend is smaller than the number of transactions. Give all egress
2083  // to the first transaction
2084  nextEgressResults_.erase(++nextEgressResults_.begin(),
2085  nextEgressResults_.end());
2086  txnMaxToSend = std::min(toSend, egressBodySizeLimit_);
2087  nextEgressResults_.front().second = 1;
2088  }
2089  if (nextEgressResults_.size() > 1 && txnMaxToSend > egressBodySizeLimit_) {
2090  // Cap the max to egressBodySizeLimit_, and recompute toSend accordingly
2091  txnMaxToSend = egressBodySizeLimit_;
2092  toSend = txnMaxToSend / nextEgressResults_.front().second;
2093  }
2094  // split allowed by relative weight, with some minimum
2095  for (auto txnPair: nextEgressResults_) {
2096  uint32_t txnAllowed = txnPair.second * toSend;
2097  if (nextEgressResults_.size() > 1) {
2098  CHECK_LE(txnAllowed, egressBodySizeLimit_);
2099  }
2100  if (connFlowControl_) {
2101  CHECK_LE(txnAllowed, connFlowControl_->getAvailableSend());
2102  }
2103  if (txnAllowed == 0) {
2104  // The ratio * toSend was so small this txn gets nothing.
2105  VLOG(4) << *this << " breaking egress loop on 0 txnAllowed";
2106  break;
2107  }
2108 
2109  VLOG(4) << *this << " egressing txnID=" << txnPair.first->getID() <<
2110  " allowed=" << txnAllowed;
2111  txnPair.first->onWriteReady(txnAllowed, txnPair.second);
2112  }
2113  nextEgressResults_.clear();
2114  // it can be empty because of HTTPTransaction rate limiting. We should
2115  // change rate limiting to clearPendingEgress while waiting.
2116  if (!writeBuf_.empty()) {
2117  break;
2118  }
2119  }
2120  *eom = false;
2121  if (byteEventTracker_) {
2122  uint64_t needed = byteEventTracker_->preSend(cork, eom, bytesWritten_);
2123  if (needed > 0) {
2124  VLOG(5) << *this << " writeBuf_.chainLength(): "
2125  << writeBuf_.chainLength() << " txnEgressQueue_.empty(): "
2126  << txnEgressQueue_.empty();
2127 
2128  if (needed < writeBuf_.chainLength()) {
2129  // split the next EOM chunk
2130  VLOG(5) << *this << " splitting " << needed << " bytes out of a "
2131  << writeBuf_.chainLength() << " bytes IOBuf";
2132  *cork = true;
2133  if (sessionStats_) {
2134  sessionStats_->recordTTLBAIOBSplitByEom();
2135  }
2136  return writeBuf_.split(needed);
2137  } else {
2138  CHECK_EQ(needed, writeBuf_.chainLength());
2139  }
2140  }
2141  }
2142 
2143  // cork if there are txns with pending egress and room to send them
2144  *cork = !txnEgressQueue_.empty() && !isConnWindowFull();
2145  return writeBuf_.move();
2146 }
