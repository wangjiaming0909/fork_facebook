25 void QPACKDecoder::decodeStreaming(
26  uint64_t streamID,
27  std::unique_ptr<folly::IOBuf> block,
28  uint32_t totalBytes,
29  HPACK::StreamingCallback* streamingCb) {
30  Cursor cursor(block.get());
31  HPACKDecodeBuffer dbuf(cursor, totalBytes, maxUncompressed_);
32  err_ = HPACK::DecodeError::NONE;
33  uint32_t largestReference = handleBaseIndex(dbuf);
34  if (largestReference > table_.getBaseIndex()) {
35  VLOG(5) << "largestReference=" << largestReference << " > baseIndex=" <<
36  table_.getBaseIndex() << ", queuing";
37  if (queue_.size() >= maxBlocking_) {
38  VLOG(2) << "QPACK queue is full size=" << queue_.size()
39  << " maxBlocking_=" << maxBlocking_;
40  err_ = HPACK::DecodeError::TOO_MANY_BLOCKING;
41  completeDecode(HeaderCodec::Type::QPACK, streamingCb, 0, 0, false);
42  } else {
43  folly::IOBufQueue q;
44  q.append(std::move(block));
45  q.trimStart(dbuf.consumedBytes());
46  enqueueHeaderBlock(streamID, largestReference, baseIndex_,
47  dbuf.consumedBytes(), q.move(),
48  totalBytes - dbuf.consumedBytes(), streamingCb);
49  }
50  } else {
51  decodeStreamingImpl(largestReference, 0, dbuf, streamingCb);
52  }
53 }

55 uint32_t QPACKDecoder::handleBaseIndex(HPACKDecodeBuffer& dbuf) {
56  uint64_t largestReference;
57  uint64_t wireLR;
58  uint64_t maxEntries = table_.getMaxEntries();
59  uint64_t modulus = 2 * maxEntries;
60  err_ = dbuf.decodeInteger(wireLR);
61  if (err_ != HPACK::DecodeError::NONE) {
62  LOG(ERROR) << "Decode error decoding largest reference err_=" << err_;
63  return 0;
64  }
65  if (wireLR == 0) {
66  largestReference = 0;
67  } else {
68  wireLR--;
69  if (wireLR >= modulus) {
70  LOG(ERROR) << "Decode error LR out of range=" << wireLR;
71  err_ = HPACK::DecodeError::INVALID_INDEX;
72  return 0;
73  }
74  auto now = table_.getBaseIndex() % modulus;
75  if (now >= wireLR + maxEntries) {
76  // LR wrapped on more time than now
77  wireLR += modulus;
78  } else if (now + maxEntries < wireLR) {
79  // now wrapped one more time than LR
80  now += modulus;
81  }
82  largestReference = wireLR + (table_.getBaseIndex() - now);
83  }
84  VLOG(5) << "Decoded largestReference=" << largestReference;
85  uint64_t delta = 0;
86  if (dbuf.empty()) {
87  LOG(ERROR) << "Invalid prefix, no delta-base";
88  err_ = HPACK::DecodeError::BUFFER_UNDERFLOW;
89  return 0;
90  }
91  bool neg = dbuf.peek() & HPACK::Q_DELTA_BASE_NEG;
92  err_ = dbuf.decodeInteger(HPACK::Q_DELTA_BASE.prefixLength, delta);
93  if (err_ != HPACK::DecodeError::NONE) {
94  LOG(ERROR) << "Decode error decoding delta base=" << err_;
95  return 0;
96  }
97  if (neg) {
98  if (delta > largestReference) {
99  LOG(ERROR) << "Invalid delta=" << delta << " largestReference="
100  << largestReference;
101  err_ = HPACK::DecodeError::INVALID_INDEX;
102  return 0;
103  }
104  baseIndex_ = largestReference - delta;
105  } else {
106  baseIndex_ = largestReference + delta;
107  }
108  VLOG(5) << "Decoded baseIndex_=" << baseIndex_;
109  return largestReference;
110 }

112 void QPACKDecoder::decodeStreamingImpl(
113  uint32_t largestReference,
114  uint32_t consumed, HPACKDecodeBuffer& dbuf,
115  HPACK::StreamingCallback* streamingCb) {
116  uint32_t emittedSize = 0;
117 
118  while (!hasError() && !dbuf.empty()) {
119  emittedSize += decodeHeaderQ(dbuf, streamingCb);
120  if (emittedSize > maxUncompressed_) {
121  LOG(ERROR) << "exceeded uncompressed size limit of "
122  << maxUncompressed_ << " bytes";
123  err_ = HPACK::DecodeError::HEADERS_TOO_LARGE;
124  break;
125  }
126  emittedSize += 2;
127  }
128 
129  bool acknowledge = largestReference != 0;
130  if (!hasError()) {
131  // lastAcked_ is only read in encodeTableStateSync, so all completed header
132  // blocks must be call encodeHeaderAck BEFORE calling encodeTableStateSync.
133  lastAcked_ = std::max(lastAcked_, largestReference);
134  }
135  completeDecode(HeaderCodec::Type::QPACK, streamingCb,
136  consumed + dbuf.consumedBytes(), emittedSize, acknowledge);
137 }

139 uint32_t QPACKDecoder::decodeHeaderQ(
140  HPACKDecodeBuffer& dbuf,
141  HPACK::StreamingCallback* streamingCb) {
142  uint8_t byte = dbuf.peek();
143  if (byte & HPACK::Q_INDEXED.code) {
144  return decodeIndexedHeaderQ(
145  dbuf, HPACK::Q_INDEXED.prefixLength, false, streamingCb, nullptr);
146  } else if (byte & HPACK::Q_LITERAL_NAME_REF.code) {
147  return decodeLiteralHeaderQ(
148  dbuf, false, true, HPACK::Q_LITERAL_NAME_REF.prefixLength, false,
149  streamingCb);
150  } else if (byte & HPACK::Q_LITERAL.code) {
151  return decodeLiteralHeaderQ(
152  dbuf, false, false, HPACK::Q_LITERAL.prefixLength, false, streamingCb);
153  } else if (byte & HPACK::Q_INDEXED_POST.code) {
154  return decodeIndexedHeaderQ(
155  dbuf, HPACK::Q_INDEXED_POST.prefixLength, true, streamingCb, nullptr);
156  } else { // Q_LITERAL_NAME_REF_POST
157  return decodeLiteralHeaderQ(
158  dbuf, false, true, HPACK::Q_LITERAL_NAME_REF_POST.prefixLength, true,
159  streamingCb);
160  }

163 HPACK::DecodeError QPACKDecoder::decodeEncoderStream(
164  std::unique_ptr<folly::IOBuf> buf) {
165  ingress_.append(std::move(buf));
166  Cursor cursor(ingress_.front());
167  HPACKDecodeBuffer dbuf(cursor, ingress_.chainLength(), maxUncompressed_);
168  VLOG(6) << "Decoding control block";
169  baseIndex_ = 0;
170  err_ = HPACK::DecodeError::NONE;
171  while (!hasError() && !dbuf.empty()) {
172  decodeEncoderStreamInstruction(dbuf);
173  if (err_ == HPACK::DecodeError::BUFFER_UNDERFLOW) {
174  ingress_.trimStart(partial_.consumed);
175  drainQueue();
176  return HPACK::DecodeError::NONE;
177  }
178  }
179  ingress_.trimStart(dbuf.consumedBytes());
180  if (hasError()) {
181  return err_;
182  } else {
183  drainQueue();
184  return HPACK::DecodeError::NONE;
185  }
186 }

188 void QPACKDecoder::decodeEncoderStreamInstruction(HPACKDecodeBuffer& dbuf) {
189  uint8_t byte = dbuf.peek();
190  partial_.consumed = dbuf.consumedBytes();
191  if (partial_.state == Partial::VALUE ||
192  byte & HPACK::Q_INSERT_NAME_REF.code) {
193  // If partial state is VALUE, it might have been a NO_NAME_REF instruction,
194  // but we've already parsed the name, so it doesn't matter
195  decodeLiteralHeaderQ(
196  dbuf, true, true, HPACK::Q_INSERT_NAME_REF.prefixLength, false,
197  nullptr);
198  } else if (byte & HPACK::Q_INSERT_NO_NAME_REF.code) {
199  decodeLiteralHeaderQ(
200  dbuf, true, false, HPACK::Q_INSERT_NO_NAME_REF.prefixLength, false,
201  nullptr);
202  } else if (byte & HPACK::Q_TABLE_SIZE_UPDATE.code) {
203  handleTableSizeUpdate(dbuf, table_);
204  } else { // must be Q_DUPLICATE=000
205  headers_t emitted;
206  decodeIndexedHeaderQ(
207  dbuf, HPACK::Q_DUPLICATE.prefixLength, false, nullptr, &emitted);
208  if (!hasError()) {
209  CHECK(!emitted.empty());
210  table_.add(std::move(emitted[0]));
211  }
212  }
213 }

215 uint32_t QPACKDecoder::decodeLiteralHeaderQ(
216  HPACKDecodeBuffer& dbuf,
217  bool indexing,
218  bool nameIndexed,
219  uint8_t prefixLength,
220  bool aboveBase,
221  HPACK::StreamingCallback* streamingCb) {
222  bool allowPartial = (streamingCb == nullptr);
223  Partial localPartial;
224  Partial* partial = (allowPartial) ? &partial_ : &localPartial;
225  if (partial->state == Partial::NAME) {
226  if (nameIndexed) {
227  uint64_t nameIndex = 0;
228  bool isStaticName = !aboveBase && (dbuf.peek() & (1 << prefixLength));
229  err_ = dbuf.decodeInteger(prefixLength, nameIndex);
230  if (allowPartial && err_ == HPACK::DecodeError::BUFFER_UNDERFLOW) {
231  return 0;
232  }
233  if (err_ != HPACK::DecodeError::NONE) {
234  LOG(ERROR) << "Decode error decoding index err_=" << err_;
235  return 0;
236  }
237  nameIndex++;
238  // validate the index
239  if (!isValid(isStaticName, nameIndex, aboveBase)) {
240  LOG(ERROR) << "received invalid index: " << nameIndex;
241  err_ = HPACK::DecodeError::INVALID_INDEX;
242  return 0;
243  }
244  partial->header.name = getHeader(
245  isStaticName, nameIndex, baseIndex_, aboveBase).name;
246  } else {
247  folly::fbstring headerName;
248  err_ = dbuf.decodeLiteral(prefixLength, headerName);
249  if (allowPartial && err_ == HPACK::DecodeError::BUFFER_UNDERFLOW) {
250  return 0;
251  }
252  if (err_ != HPACK::DecodeError::NONE) {
253  LOG(ERROR) << "Error decoding header name err_=" << err_;
254  return 0;
255  }
256  partial->header.name = headerName;
257  }
258  partial->state = Partial::VALUE;
259  partial->consumed = dbuf.consumedBytes();
260  }
261  // value
262  err_ = dbuf.decodeLiteral(partial->header.value);
263  if (allowPartial && err_ == HPACK::DecodeError::BUFFER_UNDERFLOW) {
264  return 0;
265  }
266  if (err_ != HPACK::DecodeError::NONE) {
267  LOG(ERROR) << "Error decoding header value name=" << partial->header.name
268  << " err_=" << err_;
269  return 0;
270  }
271  partial->state = Partial::NAME;
272 
273  uint32_t emittedSize = emit(partial->header, streamingCb, nullptr);
274 
275  if (indexing) {
276  table_.add(std::move(partial->header));
277  }
278 
279  return emittedSize;
280 }

282 uint32_t QPACKDecoder::decodeIndexedHeaderQ(
283  HPACKDecodeBuffer& dbuf,
284  uint32_t prefixLength,
285  bool aboveBase,
286  HPACK::StreamingCallback* streamingCb,
287  headers_t* emitted) {
288  uint64_t index;
289  bool isStatic = !aboveBase && (dbuf.peek() & (1 << prefixLength));
290  err_ = dbuf.decodeInteger(prefixLength, index);
291  if (err_ != HPACK::DecodeError::NONE) {
292  LOG(ERROR) << "Decode error decoding index err_=" << err_;
293  return 0;
294  }
295  index++;
296  // validate the index
297  if (index == 0 || !isValid(isStatic, index, aboveBase)) {
298  LOG(ERROR) << "received invalid index: " << index;
299  err_ = HPACK::DecodeError::INVALID_INDEX;
300  return 0;
301  }
302 
303  auto& header = getHeader(isStatic, index, baseIndex_, aboveBase);
304  return emit(header, streamingCb, emitted);
305 }

341 std::unique_ptr<folly::IOBuf> QPACKDecoder::encodeCancelStream(
342  uint64_t streamId) {
343  // Remove this stream from the queue
344  auto it = queue_.begin();
345  while (it != queue_.end()) {
346  if (it->second.streamID == streamId) {
347  it = queue_.erase(it);
348  } else {
349  it++;
350  }
351  }
352  HPACKEncodeBuffer ackEncoder(kGrowth, false);
353  ackEncoder.encodeInteger(streamId, HPACK::Q_CANCEL_STREAM);
354  return ackEncoder.release();

357 void QPACKDecoder::enqueueHeaderBlock(
358  uint64_t streamID,
359  uint32_t largestReference,

361  uint32_t consumed,
362  std::unique_ptr<folly::IOBuf> block,

364  HPACK::StreamingCallback* streamingCb) {
365  // TDOO: this queue is currently unbounded and has no timeouts
366  CHECK_GT(largestReference, table_.getBaseIndex());
367  queue_.emplace(
368  std::piecewise_construct,
369  std::forward_as_tuple(largestReference),
370  std::forward_as_tuple(streamID, baseIndex, length, consumed,

372  holBlockCount_++;
373  VLOG(5) << "queued block=" << largestReference << " len=" << length;
374  queuedBytes_ += length;

377 bool QPACKDecoder::decodeBlock(uint32_t largestReference,
378  const PendingBlock& pending) {
379  if (pending.length > 0) {
380  VLOG(5) << "decodeBlock len=" << pending.length;
381  folly::io::Cursor cursor(pending.block.get());
382  HPACKDecodeBuffer dbuf(cursor, pending.length, maxUncompressed_);
383  DCHECK_LE(pending.length, queuedBytes_);
384  queuedBytes_ -= pending.length;
385  baseIndex_ = pending.baseIndex;
386  folly::DestructorCheck::Safety safety(*this);
387  decodeStreamingImpl(largestReference, pending.consumed, dbuf, pending.cb);
388  // The callback way destroy this, if so stop queue processing
389  if (safety.destroyed()) {
390  return true;
391  }
392  }
393  return false;
394 }

396 void QPACKDecoder::drainQueue() {
397  auto it = queue_.begin();
398  while (!queue_.empty() && it->first <= table_.getBaseIndex() &&
399  !hasError()) {
400  auto id = it->first;
401  PendingBlock block = std::move(it->second);
402  queue_.erase(it);
403  if (decodeBlock(id, block)) {
404  return;
405  }
406  it = queue_.begin();
407  }
408 }
