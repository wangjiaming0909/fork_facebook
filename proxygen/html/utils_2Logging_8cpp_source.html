104 string IOBufPrinter::printChain(const IOBuf* buf,
105  Format format,
106  bool coalesce) {
107  uint8_t index = (uint8_t) format;
108  if (printers.size() <= index) {
109  LOG(ERROR) << "invalid format: " << index;
110  return "";

113  // empty chain
114  if (!buf) {

117

119  if (coalesce) {

123  }
124  auto b = buf;
125  string res;

128  b = b->next();
129  } while (b != buf);

134  struct stat fstat;
