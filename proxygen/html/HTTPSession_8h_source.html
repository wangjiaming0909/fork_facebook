585  template<typename... Args1, typename... Args2>
586  void invokeOnAllTransactions(void (HTTPTransaction::*fn)(Args1...),
587  Args2&&... args) {
588  DestructorGuard g(this);
589  std::vector<HTTPCodec::StreamID> ids;
590  for (const auto& txn: transactions_) {
591  ids.push_back(txn.first);
592  }
593  for (auto idit = ids.begin(); idit != ids.end() && !transactions_.empty();
594  ++idit) {
595  auto txn = findTransaction(*idit);
596  if (txn != nullptr) {
597  (txn->*fn)(std::forward<Args2>(args)...);
598  }
599  }
600  }
