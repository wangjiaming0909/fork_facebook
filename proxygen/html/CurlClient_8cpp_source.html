57 HTTPHeaders CurlClient::parseHeaders(const std::string& headersString) {
58  vector<StringPiece> headersList;
59  HTTPHeaders headers;
60  folly::split(",", headersString, headersList);
61  for (const auto& headerPair: headersList) {
62  vector<StringPiece> nv;
63  folly::split('=', headerPair, nv);
64  if (nv.size() > 0) {
65  if (nv[0].empty()) {
66  continue;
67  }
68  StringPiece value("");
69  if (nv.size() > 1) {
70  value = nv[1];
71  } // trim anything else
72  headers.add(nv[0], value);
73  }
74  }
75  return headers;
76 }

132 void CurlClient::sendRequest(HTTPTransaction* txn) {
133  txn_ = txn;
134  request_.setMethod(httpMethod_);
135  request_.setHTTPVersion(httpMajor_, httpMinor_);
136  if (proxy_) {
137  request_.setURL(url_.getUrl());
138  } else {
139  request_.setURL(url_.makeRelativeURL());
140  }
141  request_.setSecure(url_.isSecure());
142  if (h2c_) {
143  HTTP2Codec::requestUpgrade(request_);
144  }
145 
146  if (!request_.getHeaders().getNumberOfValues(HTTP_HEADER_USER_AGENT)) {
147  request_.getHeaders().add(HTTP_HEADER_USER_AGENT, "proxygen_curl");
148  }
149  if (!request_.getHeaders().getNumberOfValues(HTTP_HEADER_HOST)) {
150  request_.getHeaders().add(HTTP_HEADER_HOST, url_.getHostAndPort());
151  }
152  if (!request_.getHeaders().getNumberOfValues(HTTP_HEADER_ACCEPT)) {
153  request_.getHeaders().add("Accept", "*/*");
154  }
155  if (loggingEnabled_) {
156  request_.dumpMessage(4);
157  }
158 
159  txn_->sendHeaders(request_);
160 
161  unique_ptr<IOBuf> buf;
162  if (httpMethod_ == HTTPMethod::POST) {
163 
164  const uint16_t kReadSize = 4096;
165  ifstream inputFile(inputFilename_, ios::in | ios::binary);
166 
167  // Reading from the file by chunks
168  // Important note: It's pretty bad to call a blocking i/o function like
169  // ifstream::read() in an eventloop - but for the sake of this simple
170  // example, we'll do it.
171  // An alternative would be to put this into some folly::AsyncReader
172  // object.
173  while (inputFile.good()) {
174  buf = IOBuf::createCombined(kReadSize);
175  inputFile.read((char*)buf->writableData(), kReadSize);
176  buf->append(inputFile.gcount());
177  txn_->sendBody(move(buf));
178  }
179  }
180 
181  // note that sendBody() is called only for POST. It's fine not to call it
182  // at all.
183 
184  txn_->sendEOM();
185 }
