91 HTTPUpstreamSession::newTransaction(HTTPTransaction::Handler* handler) {
92  if (!supportsMoreTransactions() || draining_) {
93  // This session doesn't support any more parallel transactions
94  return nullptr;
95  }
96 
97  if (!started_) {
98  startNow();
99  }

103 
104  if (txn) {
105  DestructorGuard dg(this);
106  auto txnID = txn->getID();
107  txn->setHandler(CHECK_NOTNULL(handler));
108  setNewTransactionPauseState(txnID);
109  }
110  return txn;
