45 void encodeBlocks(QPACKCodec& decoder,
46  std::vector<std::vector<compress::Header>>& blocks) {
47  uint64_t streamId = 1;
48  QPACKCodec encoder;
49  encoder.setMaxVulnerable(FLAGS_max_blocking);
50  encoder.setEncoderHeaderTableSize(FLAGS_table_size);
51  folly::File outputF(FLAGS_output, O_CREAT | O_RDWR | O_TRUNC);
52  IOBufQueue outbuf;
53  QueueAppender appender(&outbuf, 1000);
54  uint64_t bytesIn = 0;
55  uint64_t bytesOut = 0;
56  for (auto& block : blocks) {
57  auto result = encoder.encode(block, streamId);
58  // always write stream before control to test decoder blocking
59  SimStreamingCallback cb(streamId, nullptr);
60  if (result.stream) {
61  decoder.decodeStreaming(streamId, result.stream->clone(),
62  result.stream->computeChainDataLength(),
63  &cb);
64  writeFrame(appender, streamId, std::move(result.stream));
65  if (FLAGS_ack && cb.acknowledge) {
66  encoder.decodeDecoderStream(decoder.encodeHeaderAck(streamId));
67  }
68  }
69  if (result.control) {
70  decoder.decodeEncoderStream(result.control->clone());
71  writeFrame(appender, 0, std::move(result.control));
72  if (FLAGS_ack) {
73  // There can be TSS when the decoder is non-blocking
74  auto res = decoder.encodeTableStateSync();
75  if (res) {
76  encoder.decodeDecoderStream(std::move(res));
77  }
78  }
79  }
80  bytesIn += encoder.getEncodedSize().uncompressed;
81  auto out = outbuf.move();
82  auto iov = out->getIov();
83  bytesOut += writevFull(outputF.fd(), iov.data(), iov.size());
84  streamId++;
85  }
86  LOG(INFO) << "Encoded " << (streamId - 1) << " streams. Bytes in="
87  << bytesIn << " Bytes out=" << bytesOut << " Ratio="
88  << int32_t(100 * (1 - (bytesOut / double(bytesIn))));
89 }

148  void onIngress(folly::IOBufQueue& inbuf) override {
149  while (true) {
150  if (state == HEADER) {
151  if (inbuf.chainLength() < (sizeof(uint64_t) + sizeof(uint32_t))) {
152  return;
153  }
154  Cursor c(inbuf.front());
155  streamId = c.readBE<uint64_t>();
156  length = c.readBE<uint32_t>();
157  inbuf.trimStart(sizeof(uint64_t) + sizeof(uint32_t));
158  state = DATA;
159  }
160  if (state == DATA) {
161  if (inbuf.chainLength() < length) {
162  return;
163  }
164  auto buf = inbuf.split(length);
165  callback(streamId, length, std::move(buf));
166  state = HEADER;
167  }

172 int decodeAndVerify(QPACKCodec& decoder, const proxygen::HTTPArchive& har) {
173  std::map<uint64_t, SimStreamingCallback> streams;
174  CompressedReader creader(
175  [&] (uint64_t streamId, uint32_t length,
176  std::unique_ptr<folly::IOBuf> buf) {
177  if (streamId == 0) {
178  CHECK_EQ(decoder.decodeEncoderStream(std::move(buf)),
179  HPACK::DecodeError::NONE);
180  } else {
181  auto res = streams.emplace(std::piecewise_construct,
182  std::forward_as_tuple(streamId),
183  std::forward_as_tuple(streamId, nullptr,
184  FLAGS_public));
185  decoder.decodeStreaming(
186  streamId, std::move(buf), length, &res.first->second);
187  }
188  });
189  if (creader.read()) {
190  return 1;
191  }
192  size_t i = 0;
193  for (const auto& req : streams) {
194  if (req.second.error != HPACK::DecodeError::NONE) {
195  LOG(ERROR) << "request=" << req.first
196  << " failed to decode error=" << req.second.error;
197  return 1;
198  }
199  if (!(req.second.msg == har.requests[i])) {
200  LOG(ERROR) << "requests are not equal, got=" << req.second.msg
201  << " expected=" << har.requests[i];
202  }
203  i++;
204  }
205  LOG(INFO) << "Verified " << i << " streams.";
206  return 0;
207 }

310  static bool iseol(uint8_t ch) {
311  return ch == '\r' || ch == '\n';
312  }
313 
314  void onIngress(folly::IOBufQueue& input) override {
315  Cursor c(input.front());
316  while (!c.isAtEnd()) {
317  switch (state_) {
318  case LINESTART:
319  {
320  seenR = false;
321  auto p = c.peek();
322  switch (p.first[0]) {
323  case '#':
324  state_ = COMMENT;
325  break;
326  case '\r':
327  case '\n':
328  if (!blocks.back().empty()) {
329  blocks.emplace_back();
330  }
331  state_ = EOL;
332  break;
333  default:
334  state_ = NAME;
335  strings.emplace_back();
336  }
337  break;
338  }
339  case COMMENT:
340  c.skipWhile([] (uint8_t ch) {
341  return !iseol(ch);
342  });
343  if (!c.isAtEnd()) {
344  state_ = EOL;
345  }
346  break;
347  case EOL:
348  {
349  auto p = c.peek();
350  if (p.first[0] == '\n') {
351  c.skip(1);
352  state_ = LINESTART;
353  } else if (seenR) { // \r followed by anything but \n -> mac newline
354  state_ = LINESTART;
355  } else if (p.first[0] == '\r') {
356  c.skip(1);
357  seenR = true;
358  }
359  break;
360  }
361  case NAME:
362  strings.back() += c.readWhile([] (uint8_t ch) {
363  return ch != '\t';
364  });
365  if (!c.isAtEnd()) {
366  c.skip(1);
367  state_ = VALUE;
368  strings.emplace_back();
369  }
370  break;
371  case VALUE:
372  strings.back() += c.readWhile([] (uint8_t ch) {
373  return !iseol(ch);
374  });
375  if (!c.isAtEnd()) {
376  CHECK_GE(strings.size(), 2);
377  blocks.back().emplace_back(
378  compress::Header::makeHeaderForTest(
379  *(strings.rbegin() + 1), *strings.rbegin()));
380  state_ = EOL;
381  }
382  break;
383  }
384  }
385  input.move();
386  }
387
