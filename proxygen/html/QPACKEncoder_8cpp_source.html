40 QPACKEncoder::EncodeResult
41 QPACKEncoder::encodeQ(const vector<HPACKHeader>& headers, uint64_t streamId) {
42  OutstandingBlock outstandingBlock;
43  // curOutstanding_ points to a local stack variable, it's mostly for
44  // convenience so other methods invoked from here can access it.
45  curOutstanding_ = &outstandingBlock;
46  auto baseIndex = table_.getBaseIndex();
47 
48  uint32_t largestReference = 0;
49  for (const auto& header: headers) {
50  encodeHeaderQ(header, baseIndex, &largestReference);
51  }
52 
53  auto streamBlock = streamBuffer_.release();
54 
55  // encode the prefix
56  if (largestReference == 0) {
57  streamBuffer_.encodeInteger(0); // LR
58  streamBuffer_.encodeInteger(0); // baseIndex
59  } else {
60  auto wireLR = (largestReference % (2 * table_.getMaxEntries())) + 1;
61  streamBuffer_.encodeInteger(wireLR);
62  if (largestReference > baseIndex) {
63  streamBuffer_.encodeInteger(largestReference - baseIndex,
64  HPACK::Q_DELTA_BASE_NEG,
65  HPACK::Q_DELTA_BASE.prefixLength);
66  } else {
67  streamBuffer_.encodeInteger(baseIndex - largestReference,
68  HPACK::Q_DELTA_BASE_POS,
69  HPACK::Q_DELTA_BASE.prefixLength);
70  }
71  }
72  auto streamBuffer = streamBuffer_.release();
73  streamBuffer->prependChain(std::move(streamBlock));
74 
75  auto controlBuf = controlBuffer_.release();
76  // curOutstanding_.references could be empty, if the block encodes only static
77  // headers and/or literals. If so we don't track anything.
78  if (!curOutstanding_->references.empty()) {
79  if (curOutstanding_->vulnerable) {
80  DCHECK(allowVulnerable());
81  numVulnerable_++;
82  }
83  outstanding_[streamId].emplace_back(std::move(outstandingBlock));
84  }
85  // Clear the pointer to our stack
86  curOutstanding_ = nullptr;
87 
88  return { std::move(controlBuf), std::move(streamBuffer) };
89 }

91 void QPACKEncoder::encodeHeaderQ(
92  const HPACKHeader& header, uint32_t baseIndex, uint32_t* largestReference) {
93  uint32_t index = getStaticTable().getIndex(header);
94  if (index > 0) {
95  // static reference
96  streamBuffer_.encodeInteger(index - 1,
97  HPACK::Q_INDEXED.code | HPACK::Q_INDEXED_STATIC,
98  HPACK::Q_INDEXED.prefixLength);
99  return;
100  }
101 
102  bool indexable = shouldIndex(header);
103  if (indexable) {
104  index = table_.getIndex(header, allowVulnerable());
105  if (index == QPACKHeaderTable::UNACKED) {
106  index = 0;
107  indexable = false;
108  }
109  }
110  if (index != 0) {
111  // dynamic reference
112  bool duplicated = false;
113  std::tie(duplicated, index) = maybeDuplicate(index);
114  // index is now 0 or absolute
115  indexable &= (duplicated && index == 0);
116  }
117  if (index == 0) {
118  // No valid entry matching header, see if there's a matching name
119  uint32_t nameIndex = 0;
120  uint32_t absoluteNameIndex = 0;
121  bool isStaticName = false;
122  std::tie(isStaticName, nameIndex, absoluteNameIndex) =
123  getNameIndexQ(header.name);
124 
125  // Now check if we should emit an insertion on the control stream
126  // Don't try to index if we're out of encoder flow control
127  indexable &= maxEncoderStreamBytes_ > 0;
128  if (indexable && table_.canIndex(header)) {
129  encodeInsertQ(header, isStaticName, nameIndex);
130  CHECK(table_.add(header.copy()));
131  if (allowVulnerable() && lastEntryAvailable()) {
132  index = table_.getBaseIndex();
133  } else {
134  index = 0;
135  if (absoluteNameIndex > 0 &&
136  !table_.isValid(table_.absoluteToRelative(absoluteNameIndex))) {
137  // The insert may have invalidated the name index.
138  isStaticName = true;
139  nameIndex = 0;
140  absoluteNameIndex = 0;
141  }
142  }
143  }
144 
145  if (index == 0) {
146  // Couldn't insert it: table full, not indexable, or table contains
147  // vulnerable reference. Encode a literal on the request stream.
148  encodeStreamLiteralQ(header, isStaticName, nameIndex, absoluteNameIndex,
149  baseIndex, largestReference);
150  return;
151  }
152  }
153 
154  // Encoding a dynamic index reference
155  DCHECK_NE(index, 0);
156  trackReference(index, largestReference);
157  if (index > baseIndex) {
158  streamBuffer_.encodeInteger(index - baseIndex - 1, HPACK::Q_INDEXED_POST);
159  } else {
160  streamBuffer_.encodeInteger(baseIndex - index, HPACK::Q_INDEXED);
161  }
162 }

169 std::pair<bool, uint32_t> QPACKEncoder::maybeDuplicate(
170  uint32_t relativeIndex) {
171  auto res = table_.maybeDuplicate(relativeIndex, allowVulnerable());
172  if (res.first) {
173  VLOG(4) << "Encoded duplicate index=" << relativeIndex;
174  encodeDuplicate(relativeIndex);
175  // Note we will emit duplications even when we are out of flow control,
176  // but we won't reference them (eg: like we were at vulnerable max).
177  if (!lastEntryAvailable()) {
178  VLOG(4) << "Duplicate is not usable because it overran encoder flow "
179  "control";
180  return {true, 0};
181  }
182  }
183  return res;
184 }

186 std::tuple<bool, uint32_t, uint32_t> QPACKEncoder::getNameIndexQ(
187  const HPACKHeaderName& headerName) {
188  uint32_t absoluteNameIndex = 0;
189  uint32_t nameIndex = getStaticTable().nameIndex(headerName);
190  bool isStatic = true;
191  if (nameIndex == 0) {
192  // check dynamic table
193  nameIndex = table_.nameIndex(headerName, allowVulnerable());
194  if (nameIndex != 0) {
195  absoluteNameIndex = maybeDuplicate(nameIndex).second;
196  if (absoluteNameIndex) {
197  isStatic = false;
198  nameIndex = table_.absoluteToRelative(absoluteNameIndex);
199  } else {
200  nameIndex = 0;
201  absoluteNameIndex = 0;
202  }
203  }
204  }
205  return std::tuple<bool, uint32_t, uint32_t>(
206  isStatic, nameIndex, absoluteNameIndex);
207 }

209 void QPACKEncoder::encodeStreamLiteralQ(
210  const HPACKHeader& header, bool isStaticName, uint32_t nameIndex,
211  uint32_t absoluteNameIndex, uint32_t baseIndex, uint32_t* largestReference) {
212  if (absoluteNameIndex > 0) {
213  // Dynamic name reference, vulnerability checks already done
214  CHECK(absoluteNameIndex <= baseIndex || allowVulnerable());
215  trackReference(absoluteNameIndex, largestReference);
216  }
217  if (absoluteNameIndex > baseIndex) {
218  encodeLiteralQ(header,
219  false, /* not static */
220  true, /* post base */
221  absoluteNameIndex - baseIndex,
222  HPACK::Q_LITERAL_NAME_REF_POST);
223  } else {
224  encodeLiteralQ(header,
225  isStaticName,
226  false, /* not post base */
227  isStaticName ? nameIndex : baseIndex - absoluteNameIndex + 1,
228  HPACK::Q_LITERAL_NAME_REF);
229  }

232 void QPACKEncoder::trackReference(uint32_t absoluteIndex,
233  uint32_t* largestReference) {
234  CHECK_NE(absoluteIndex, 0);
235  CHECK(curOutstanding_);
236  if (absoluteIndex > *largestReference) {
237  *largestReference = absoluteIndex;
238  if (table_.isVulnerable(absoluteIndex)) {
239  curOutstanding_->vulnerable = true;
240  }
241  }
242  auto res = curOutstanding_->references.insert(absoluteIndex);
243  if (res.second) {
244  VLOG(5) << "Bumping refcount for absoluteIndex=" << absoluteIndex;
245  table_.addRef(absoluteIndex);
246  }
247 }

277 uint32_t QPACKEncoder::encodeLiteralQHelper(
278  HPACKEncodeBuffer& buffer,
279  const HPACKHeader& header,
280  bool isStaticName,
281  uint32_t nameIndex,
282  uint8_t staticFlag,
283  const HPACK::Instruction& idxInstr,
284  const HPACK::Instruction& litInstr) {
285  uint32_t encoded = 0;
286  // name
287  if (nameIndex) {
288  VLOG(10) << "encoding name index=" << nameIndex;
289  DCHECK_NE(nameIndex, QPACKHeaderTable::UNACKED);
290  nameIndex -= 1; // we already know it's not 0
291  uint8_t byte = idxInstr.code;
292  if (isStaticName) {
293  byte |= staticFlag;
294  }
295  encoded += buffer.encodeInteger(nameIndex, byte, idxInstr.prefixLength);
296  } else {
297  encoded += buffer.encodeLiteral(litInstr.code, litInstr.prefixLength,
298  header.name.get());
299  }
300  // value
301  encoded += buffer.encodeLiteral(header.value);
302  return encoded;
303 }

305 HPACK::DecodeError QPACKEncoder::decodeDecoderStream(
306  std::unique_ptr<folly::IOBuf> buf) {
307  decoderIngress_.append(std::move(buf));
308  folly::io::Cursor cursor(decoderIngress_.front());
309  HPACKDecodeBuffer dbuf(cursor, decoderIngress_.chainLength(), 0);
310  HPACK::DecodeError err = HPACK::DecodeError::NONE;
311  uint32_t consumed = 0;
312  while (err == HPACK::DecodeError::NONE && !dbuf.empty()) {
313  consumed = dbuf.consumedBytes();
314  auto byte = dbuf.peek();
315  if (byte & HPACK::Q_HEADER_ACK.code) {
316  err = decodeHeaderAck(dbuf, HPACK::Q_HEADER_ACK.prefixLength, false);
317  } else if (byte & HPACK::Q_CANCEL_STREAM.code) {
318  err = decodeHeaderAck(dbuf, HPACK::Q_CANCEL_STREAM.prefixLength, true);
319  } else { // TABLE_STATE_SYNC
320  uint64_t inserts = 0;
321  err = dbuf.decodeInteger(HPACK::Q_TABLE_STATE_SYNC.prefixLength, inserts);
322  if (err == HPACK::DecodeError::NONE) {
323  err = onTableStateSync(inserts);
324  } else if (err != HPACK::DecodeError::BUFFER_UNDERFLOW) {
325  LOG(ERROR) << "Failed to decode num inserts, err=" << err;
326  }
327  }
328  } // while
329  if (err == HPACK::DecodeError::BUFFER_UNDERFLOW) {
330  err = HPACK::DecodeError::NONE;
331  decoderIngress_.trimStart(consumed);
332  } else {
333  decoderIngress_.trimStart(dbuf.consumedBytes());
334  }
335  return err;
336 }

358 HPACK::DecodeError QPACKEncoder::onHeaderAck(uint64_t streamId, bool all) {
359  auto it = outstanding_.find(streamId);
360  if (it == outstanding_.end()) {
361  if (!all) {
362  LOG(ERROR) << "Received an ack with no outstanding header blocks stream="
363  << streamId;
364  return HPACK::DecodeError::INVALID_ACK;
365  } else {
366  // all implies a reset, meaning it's not an error if there are no
367  // outstanding blocks
368  return HPACK::DecodeError::NONE;
369  }
370  }
371  DCHECK(!it->second.empty()) << "Invariant violation: no blocks in stream "
372  "record";
373  VLOG(5) << ((all) ? "onCancelStream" : "onHeaderAck") << " streamId="
374  << streamId;
375  if (all) {
376  // Happens when a stream is reset
377  for (auto& block: it->second) {
378  for (auto i: block.references) {
379  VLOG(5) << "Decrementing refcount for absoluteIndex=" << i;
380  table_.subRef(i);
381  }
382  if (block.vulnerable) {
383  numVulnerable_--;
384  }
385  }
386  it->second.clear();
387  } else {
388  auto block = std::move(it->second.front());
389  it->second.pop_front();
390  // a different stream, sub all the references
391  for (auto i: block.references) {
392  VLOG(5) << "Decrementing refcount for absoluteIndex=" << i;
393  table_.subRef(i);
394  }
395  if (block.vulnerable) {
396  numVulnerable_--;
397  }
398  // largest reference is implicitly acknowledged
399  if (!block.references.empty()) {
400  auto largestReference = *block.references.rbegin();
401  VLOG(5) << "Implicitly acknowledging absoluteIndex=" << largestReference;
402  table_.setMaxAcked(largestReference);
403  }
404  }
405  if (it->second.empty()) {
406  outstanding_.erase(it);
407  }
408  return HPACK::DecodeError::NONE;
409 }
