240  HTTPHeaders& strippedHeaders) {
241  bool transferred = false;

243  name.size());
244  if (code == HTTP_HEADER_OTHER) {
245  ITERATE_OVER_STRINGS(name, {

250  codes_[pos] = HTTP_HEADER_NONE;

253  });
254  } else { // code != HTTP_HEADER_OTHER
255  ITERATE_OVER_CODES(code, {

258  strippedHeaders.headerValues_.push_back(headerValues_[pos]);

267 void
268 HTTPHeaders::stripPerHopHeaders(HTTPHeaders& strippedHeaders) {
269  int len;
270  forEachValueOfHeader(HTTP_HEADER_CONNECTION, [&]
271  (const string& stdStr) -> bool {
272  // Remove all headers specified in Connection header
273  // look for multiple values separated by commas
274  char const* str = stdStr.c_str();
275 
276  // skip leading whitespace
277  while (isLWS(*str)) str++;
278 
279  while (*str != 0) {
280  char const* pos = strchr(str, ',');
281  if (pos == nullptr) {
282  // last (or only) token, done
283 
284  // count chars in the token
285  len = 0;
286  while (str[len] != 0 && !isLWS(str[len])) len++;
287  if (len > 0) {
288  string hdr(str, len);
289  if (transferHeaderIfPresent(hdr, strippedHeaders)) {
290  VLOG(3) << "Stripped connection-named hop-by-hop header " << hdr;
291  }
292  }
293  break;
294  }
295  len = pos - str;
296  // strip trailing whitespace
297  while (len > 0 && isLWS(str[len - 1])) len--;
298  if (len > 0) {
299  // non-empty token
300  string hdr(str, len);
301  if (transferHeaderIfPresent(hdr, strippedHeaders)) {
302  VLOG(3) << "Stripped connection-named hop-by-hop header " << hdr;
303  }
304  } // else empty token, no-op
305  str = pos + 1;
306 
307  // skip whitespace
308  while (isLWS(*str)) str++;
309  }
310  return false; // continue processing "connection" headers
311  });
312 
313  // Strip hop-by-hop headers
314  auto& perHopHeaders = perHopHeaderCodes();
315  for (size_t i = 0; i < codes_.size(); ++i) {
316  if (perHopHeaders[codes_[i]]) {
317  strippedHeaders.codes_.push_back(codes_[i]);
318  strippedHeaders.headerNames_.push_back(headerNames_[i]);
319  strippedHeaders.headerValues_.push_back(headerValues_[i]);
320  codes_[i] = HTTP_HEADER_NONE;
321  ++deletedCount_;
322  VLOG(5) << "Stripped hop-by-hop header " << *headerNames_[i];
323  }
324  }
325 }
