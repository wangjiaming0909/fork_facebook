67 HTTP2PriorityQueue::Node::addChildren(list<unique_ptr<Node>>&& children) {
68  list<unique_ptr<Node>> emptyChilden;
69  uint64_t totalEnqueuedWeight = 0;
70  for (auto& child: children) {
71  if (child->inEgressTree()) {
72  totalEnqueuedWeight += child->weight_;

74  CHECK(!child->enqueuedHook_.is_linked());
75  addEnqueuedChild(child.get());

77  CHECK(!child->enqueuedHook_.is_linked());

82  if (totalEnqueuedWeight > 0) {

85  }

177 void
178 HTTP2PriorityQueue::Node::propagatePendingEgressSignal(
179  HTTP2PriorityQueue::Node *node) {
180  Node* parent = node->parent_;
181  bool stop = node->totalEnqueuedWeight_ > 0;
182  // Continue adding node->weight_ to parent_->totalEnqueuedWeight_ as
183  // long as node state changed from no-egress-in-subtree to
184  // egress-in-subtree
185  while (parent && !stop) {
186  stop = parent->inEgressTree();
187  parent->totalEnqueuedWeight_ += node->weight_;
188  parent->addEnqueuedChild(node);
189  node = parent;
190  parent = parent->parent_;
191  }

231 // Removes the node from the tree
232 void
233 HTTP2PriorityQueue::Node::removeFromTree() {
234  if (!children_.empty()) {
235  // update child weights so they sum to (approximately) this node's weight.
236  double r = double(weight_) / totalChildWeight_;
237  for (auto& child: children_) {
238  uint64_t newWeight = std::max(uint64_t(child->weight_ * r), uint64_t(1));
239  CHECK_LE(newWeight, 256);
240  child->updateWeight(uint8_t(newWeight) - 1);
241  }
242  }
243 
244  CHECK(!isEnqueued());
245  if (inEgressTree()) {
246  // Gah this is tricky.
247  // The children of this node are moving to this node's parent. We need the
248  // tree in a consistent state before calling addChildren, so mark the
249  // current node's totalEnqueuedWeight_ as 0 and propagate the clear upwards.
250  // addChildren will handle re-signalling egress.
251  totalEnqueuedWeight_ = 0;
252  propagatePendingEgressClear(this);
253  }
254 
255  // move my children to my parent
256  parent_->addChildren(std::move(children_));
257  (void)parent_->detachChild(this);
258 }

260 bool
261 HTTP2PriorityQueue::Node::iterate(
262  const std::function<bool(HTTPCodec::StreamID,
263  HTTPTransaction *, double)>& fn,
264  const std::function<bool()>& stopFn, bool all) {
265  bool stop = false;
266  if (stopFn()) {
267  return true;
268  }
269 #ifndef NDEBUG
270  CHECK_EQ(totalEnqueuedWeight_, totalEnqueuedWeightCheck_);
271 #endif
272  if (parent_ /* exclude root */ && (all || isEnqueued())) {
273  stop = fn(id_, txn_, getRelativeWeight());
274  }
275  for (auto& child: children_) {
276  if (stop || stopFn()) {
277  return true;
278  }
279  stop = child->iterate(fn, stopFn, all);
280  }
281  return stop;
282 }

284 bool
285 HTTP2PriorityQueue::Node::visitBFS(
286  double relativeParentWeight,
287  const std::function<bool(HTTP2PriorityQueue& queue, HTTPCodec::StreamID,
288  HTTPTransaction *, double)>& fn,
289  bool all,
290  PendingList& pendingNodes, bool enqueuedChildren) {
291  bool invoke = (parent_ != nullptr && (all || isEnqueued()));
292  auto relativeEnqueuedWeight = getRelativeEnqueuedWeight();
293 
294 #ifndef NDEBUG
295  CHECK_EQ(totalEnqueuedWeight_, totalEnqueuedWeightCheck_);
296 #endif
297  // Add children when all==true, or for any not invoked node with
298  // pending children
299  if (all || (!invoke && totalEnqueuedWeight_ > 0)) {
300  double newRelWeight = relativeParentWeight * relativeEnqueuedWeight;
301  if (enqueuedChildren) {
302  for (auto child = enqueuedChildren_.begin();
303  child != enqueuedChildren_.end();
304  child++) {
305  pendingNodes.emplace_back(child->id_, &(*child), newRelWeight);
306  }
307  } else {
308  for (auto& child: children_) {
309  pendingNodes.emplace_back(child->id_, child.get(), newRelWeight);
310  }
311  }
312  }
313 
314  // Invoke fn last in case it deletes this node
315  if (invoke && fn(queue_, id_, txn_,
316  relativeParentWeight * relativeEnqueuedWeight)) {
317  return true;
318  }
319 
320  return false;
321 }

457 HTTP2PriorityQueue::Handle
458 HTTP2PriorityQueue::addTransaction(HTTPCodec::StreamID id,
459  http2::PriorityUpdate pri,
460  HTTPTransaction *txn,
461  bool permanent,
462  uint64_t* depth) {
463  CHECK_NE(id, rootNodeId_);
464  CHECK_NE(id, pri.streamDependency) << "Tried to create a loop in the tree";
465  CHECK(!txn || !permanent);
466  Node *existingNode = find(id, depth);
467  if (existingNode) {
468  CHECK(!permanent);
469  existingNode->convertVirtualNode(CHECK_NOTNULL(txn));
470  updatePriority(existingNode, pri);
471  return existingNode;
472  }
473  if (!txn) {
474  if (numVirtualNodes_ >= maxVirtualNodes_) {
475  return nullptr;
476  }
477  numVirtualNodes_++;
478  }
479 
480  Node* parent = &root_;
481  if (depth) {
482  *depth = 1;
483  }
484  if (pri.streamDependency != rootNodeId_) {
485  Node* dep = find(pri.streamDependency, depth);
486  if (dep == nullptr) {
487  // specified a missing parent (timed out an idle node)?
488  VLOG(4) << "assigning default priority to txn=" << id;
489  // No point to try to instantiate one more virtual node
490  // if we already reached the virtual node limit
491  if (numVirtualNodes_ < maxVirtualNodes_) {
492  // The parent node hasn't arrived yet. For now setting
493  // its priority fields to default.
494  parent = dynamic_cast<Node*>(
495  addTransaction(pri.streamDependency,
496  {rootNodeId_,
497  http2::DefaultPriority.exclusive,
498  http2::DefaultPriority.weight},
499  nullptr,
500  permanent,
501  depth));
502  CHECK_NOTNULL(parent);
503  if (depth) {
504  *depth += 1;
505  }
506  } else {
507  VLOG(4) << "Virtual node limit reached, ignoring stream dependency "
508  << pri.streamDependency << " for new node ID " << id;
509  }
510  } else {
511  parent = dep;
512  if (depth) {
513  *depth += 1;
514  }
515  }
516  }
517  VLOG(4) << "Adding id=" << id << " with parent=" << parent->getID() <<
518  " and weight=" << ((uint16_t)pri.weight + 1);
519  auto node = std::make_unique<Node>(*this, parent, id, pri.weight, txn);
520  if (permanent) {
521  node->setPermanent();
522  } else if (!txn) {
523  scheduleNodeExpiration(node.get());
524  }
525  auto result = parent->emplaceNode(std::move(node), pri.exclusive);
526  pendingWeightChange_ = true;
527  return result;
528 }

530 HTTP2PriorityQueue::Handle
531 HTTP2PriorityQueue::updatePriority(HTTP2PriorityQueue::Handle handle,
532  http2::PriorityUpdate pri,
533  uint64_t* depth) {
534  Node* node = CHECK_NOTNULL(dynamic_cast<HTTP2PriorityQueue::Node*>(handle));
535  pendingWeightChange_ = true;
536  VLOG(4) << "Updating id=" << node->getID() << " with parent=" <<
537  pri.streamDependency << " and weight=" << ((uint16_t)pri.weight + 1);
538  node->updateWeight(pri.weight);
539  CHECK_NE(pri.streamDependency, node->getID()) <<
540  "Tried to create a loop in the tree";
541  if (pri.streamDependency == node->parentID() && !pri.exclusive) {
542  // no move
543  if (depth) {
544  *depth = handle->calculateDepth();
545  }
546  return handle;
547  }
548 
549  Node* newParent = find(pri.streamDependency, depth);
550  if (!newParent) {
551  if (pri.streamDependency == rootNodeId_ ||
552  numVirtualNodes_ >= maxVirtualNodes_) {
553  newParent = &root_;
554  } else {
555  // allocate a virtual node for non-existing parent in my depenency tree
556  // then do normal priority processing
557  newParent = dynamic_cast<Node*>(
558  addTransaction(pri.streamDependency,
559  {rootNodeId_,
560  http2::DefaultPriority.exclusive,
561  http2::DefaultPriority.weight},
562  nullptr,
563  false,
564  depth));
565 
566  CHECK_NOTNULL(newParent);
567  VLOG(4) << "updatePriority missing parent, creating virtual parent="
568  << newParent->getID() << " for txn=" << node->getID();
569  }
570  }
571 
572  if (newParent->isDescendantOf(node)) {
573  newParent = newParent->reparent(node->getParent(), false);
574  }
575  node = node->reparent(newParent, pri.exclusive);
576  if (depth) {
577  *depth = node->calculateDepth();
578  }
579  return node;
580 }

620 void
621 HTTP2PriorityQueue::iterateBFS(
622  const std::function<bool(HTTP2PriorityQueue&, HTTPCodec::StreamID,
623  HTTPTransaction *, double)>& fn,
624  const std::function<bool()>& stopFn, bool all) {
625  Node::PendingList pendingNodes{{rootNodeId_, &root_, 1.0}};
626  Node::PendingList newPendingNodes;
627  bool stop = false;
628 
629  updateEnqueuedWeight();
630  while (!stop && !stopFn() && !pendingNodes.empty()) {
631  CHECK(newPendingNodes.empty());
632  while (!stop && !pendingNodes.empty()) {

634  if (node) {
635  stop = node->visitBFS(pendingNodes.front().ratio, fn, all,
636  newPendingNodes, false /* all children */);
637  }
638  pendingNodes.pop_front();
639  }
640  std::swap(pendingNodes, newPendingNodes);
641  }

652 void
653 HTTP2PriorityQueue::nextEgress(HTTP2PriorityQueue::NextEgressResult& result,
654  bool spdyMode) {
655  struct WeightCmp {
656  bool operator()(const std::pair<HTTPTransaction*, double>& t1,
657  const std::pair<HTTPTransaction*, double>& t2) {
658  return t1.second > t2.second;
659  }
660  };
661 
662  result.reserve(activeCount_);
663  nextEgressResults_ = &result;
664 
665  updateEnqueuedWeight();
666  Node::PendingList pendingNodes;
667  Node::PendingList pendingNodesTmp;
668  pendingNodes.emplace_back(rootNodeId_, &root_, 1.0);
669  bool stop = false;
670  do {
671  while (!stop && !pendingNodes.empty()) {
672  Node* node = pendingNodes.front().node;
673  if (node) {
674  stop = node->visitBFS(pendingNodes.front().ratio, nextEgressResult,
675  false, pendingNodesTmp,
676  true /* enqueued children */);
677  }
678  pendingNodes.pop_front();
679  }
680  // In SPDY mode, we stop as soon one level of the tree produces results,
681  // then normalize the ratios.
682  if (spdyMode && !result.empty() && !pendingNodesTmp.empty()) {
683  double totalRatio = 0;
684  for (auto &txnPair: result) {
685  totalRatio += txnPair.second;
686  }
687  CHECK_GT(totalRatio, 0);
688  for (auto &txnPair: result) {
689  txnPair.second = txnPair.second / totalRatio;
690  }
691  break;
692  }
693  std::swap(pendingNodes, pendingNodesTmp);
694  } while (!stop && !pendingNodes.empty());
695  std::sort(result.begin(), result.end(), WeightCmp());
696  nextEgressResults_ = nullptr;
697 }

700 HTTP2PriorityQueue::find(HTTPCodec::StreamID id, uint64_t* depth) {
701  if (id == rootNodeId_) {
702  return nullptr;
703  }
704  auto it = nodes_.find(id, Node::IdHash(), Node::IdNodeEqual());
705  if (it == nodes_.end()) {
706  return nullptr;
