573  /* Setter and getter for the SPDY priority value (0 - 7). When serialized
574  * to SPDY/2, Codecs will collpase 0,1 -> 0, 2,3 -> 1, etc.
575  *
576  * Negative values of pri are interpreted much like negative array
577  * indexes in python, so -1 will be the largest numerical priority
578  * value for this SPDY version (i.e. 3 for SPDY/2 or 7 for SPDY/3),
579  * -2 the second largest (i.e. 2 for SPDY/2 or 6 for SPDY/3).
580  */
581  const static int8_t kMaxPriority;
582 
583  static uint8_t normalizePriority(int8_t pri) {
584  if (pri > kMaxPriority || pri < -kMaxPriority) {
585  // outside [-7, 7] => highest priority
586  return kMaxPriority;
587  } else if (pri < 0) {
588  return pri + kMaxPriority + 1;
589  }
590  return pri;
591  }
