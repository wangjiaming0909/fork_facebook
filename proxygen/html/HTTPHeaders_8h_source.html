302 template <typename T> // T = string
303 void HTTPHeaders::add(folly::StringPiece name, T&& value) {
304  assert(name.size());
305  const HTTPHeaderCode code = HTTPCommonHeaders::hash(name.data(), name.size());
306  codes_.push_back(code);
307  headerNames_.push_back((code == HTTP_HEADER_OTHER)

309  : HTTPCommonHeaders::getPointerToHeaderName(code));
310  headerValues_.emplace_back(std::forward<T>(value));

320 // iterate over the positions (in vector) of all headers with given code
321 #define ITERATE_OVER_CODES(Code, Block) \
322  { \
323  const HTTPHeaderCode* ptr = codes_.data(); \
324  while (ptr) { \
325  ptr = (HTTPHeaderCode*)memchr( \
326  (void*)ptr, (Code), codes_.size() - (ptr - codes_.data())); \
327  if (ptr == nullptr) \
328  break; \
329  const size_t pos = ptr - codes_.data(); \
330  {Block} ptr++; \
331  } \
332  }
333 
334 // iterate over the positions of all headers with given name
335 #define ITERATE_OVER_STRINGS(String, Block) \
336  ITERATE_OVER_CODES(HTTP_HEADER_OTHER, { \
337  if (caseInsensitiveEqual((String), *headerNames_[pos])) { \
338  {Block} \
339  } \
340 })
341 
342 // iterate over the positions of all headers with given name ignoring - and _
343 #define ITERATE_OVER_STRINGS_ALL_VERSION(String, Block) \
344  ITERATE_OVER_CODES(HTTP_HEADER_OTHER, { \
345  if (caseUnderscoreInsensitiveEqual((String), *headerNames_[pos])) { \
346  {Block} \

412 bool HTTPHeaders::removeByPredicate(LAMBDA func) {
413  bool removed = false;
414  for (size_t i = 0; i < codes_.size(); ++i) {
415  if (codes_[i] == HTTP_HEADER_NONE ||
416  !func(codes_[i], *headerNames_[i], headerValues_[i])) {
417  continue;
418  }
419 
420  if (codes_[i] == HTTP_HEADER_OTHER) {
421  delete headerNames_[i];
422  headerNames_[i] = nullptr;
423  }
424 
425  codes_[i] = HTTP_HEADER_NONE;
426  ++deletedCount_;
427  removed = true;
428  }
429 
430  return removed;
431 }
